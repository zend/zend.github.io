发个带颜色的printf：  
enum zh\_color\_set {  
COLOR_BLACK = 30,  
COLOR_RED,  
COLOR_GREEN,  
COLOR_YELLOW,  
COLOR_BLUE,  
COLOR_MAGENTA,  
COLOR_CYAN,  
COLOR_WHITE  
};  
void zh_printf(int color, char *fmt, &#8230;)  
{  
va_list ap;  
printf(&#8220;%c[1;%d;40m&#8221;, 0x1B, color);

va_start(ap, fmt);  
vprintf(fmt, ap);  
va_end(ap);

printf(&#8220;%c[0m&#8221;, 0x1B);  
}
